YACK (Yet Another Currying Kata) 
            
function yack(fn, ...args) {
  let foo = (...args2) => {
    return args2.length >= fn.length
    ? fn(...args2)
    : (...args3) => foo(...args2, ...args3)
  }

  return foo(...args)
}
